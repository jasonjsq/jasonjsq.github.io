牛客-排好序的矩阵找数

#include
using namespace std;

int main()
{
long n,m,k;
cin>>n;
cin>>m;
cin>>k;
long matrix[n][m];

for(int i=0;i<n;i++)
{
    for(int j=0;j<m;j++)
    {
        cin>>matrix[i][j];
    }
}
int mark=0;
long i=0;long j=m-1;
while(i<n&&j>=0)
{
    if(matrix[i][j]==k)
    {
        mark=1;break;
    }
    if(matrix[i][j]>k)
    {
        j--;
    }
    else
    {
        i++;
    }
}
if(mark)
{
    cout<<"Yes";
}
else{
    cout<<"No";
}

}
